```js live removeFragment
() => {
  const [value, setValue] = React.useState('center');

  return (
    <ToggleButtonGroup value={value} onChange={value => setValue(value)}>
      <ToggleButton value="start">Start</ToggleButton>
      <ToggleButton value="center">Center</ToggleButton>
      <ToggleButton value="end">End</ToggleButton>
    </ToggleButtonGroup>
  );
};
```

## Examples

### Variants

There are 2 variants to choose from.

```js live removeFragment
() => {
  const [value, setValue] = React.useState('center');

  return (
    <>
      <ToggleButtonGroup value={value} onChange={value => setValue(value)}>
        <ToggleButton value="start" variant="normal">
          Start
        </ToggleButton>
        <ToggleButton value="center" variant="normal">
          Center
        </ToggleButton>
        <ToggleButton value="end" variant="normal">
          End
        </ToggleButton>
      </ToggleButtonGroup>

      <ToggleButtonGroup
        value={value}
        onChange={value => setValue(value)}
        mt={6}
      >
        <ToggleButton value="start" variant="fade">
          Start
        </ToggleButton>
        <ToggleButton value="center" variant="fade">
          Center
        </ToggleButton>
        <ToggleButton value="end" variant="fade">
          End
        </ToggleButton>
      </ToggleButtonGroup>
    </>
  );
};
```

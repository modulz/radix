---
title: Popover
component: Popover
description: An accessible Popover component that displays rich content as a result of clicking on a target.
---

```js live removeFragment
() => {
  const [isOpen, setIsOpen] = React.useState(false);
  const buttonRef = React.useRef(null);

  return (
    <>
      <Button ref={buttonRef} onClick={() => setIsOpen(true)}>
        Open Popover
      </Button>
      <Popover
        targetRef={buttonRef}
        isOpen={isOpen}
        onClose={() => setIsOpen(false)}
      >
        <Box p={4}>Popover content</Box>
      </Popover>
    </>
  );
};
```

## Examples

### Side

There are 4 side options to choose from.

```js live removeFragment
() => {
  const [isOpen, setIsOpen] = React.useState(false);
  const [side, setSide] = React.useState('right');
  const buttonRef = React.useRef(null);

  return (
    <>
      <Button ref={buttonRef} onClick={() => setIsOpen(true)}>
        Open Popover
      </Button>
      <Popover
        targetRef={buttonRef}
        isOpen={isOpen}
        onClose={() => setIsOpen(false)}
        side={side}
      >
        <Box p={3}>
          <Heading size={0} mb={3} fontWeight={500}>
            Pick a side
          </Heading>
          <Select value={side} onValueChange={setSide}>
            <Option value="top" label="top" />
            <Option value="right" label="right" />
            <Option value="bottom" label="bottom" />
            <Option value="left" label="left" />
          </Select>
        </Box>
      </Popover>
    </>
  );
};
```

### Side offset

Control the side offset distance.

```js live removeFragment
() => {
  const [isOpen, setIsOpen] = React.useState(false);
  const [sideOffset, setSideOffset] = React.useState(0);
  const buttonRef = React.useRef(null);

  return (
    <>
      <Button ref={buttonRef} onClick={() => setIsOpen(true)}>
        Open Popover
      </Button>
      <Popover
        targetRef={buttonRef}
        isOpen={isOpen}
        onClose={() => setIsOpen(false)}
        side="right"
        sideOffset={Number(sideOffset)}
      >
        <Box p={3}>
          <Heading size={0} mb={3} fontWeight={500}>
            Set an side offset
          </Heading>
          <Slider
            min="-10"
            max="10"
            value={sideOffset}
            onChange={event => setSideOffset(event.target.value)}
          />
        </Box>
      </Popover>
    </>
  );
};
```
